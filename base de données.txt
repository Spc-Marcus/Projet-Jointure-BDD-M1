CREATE TABLE Details (
    OrderID VARCHAR(10) PRIMARY KEY,
    Amount INT,
    Profit INT,
    Quantity INT,
    Category VARCHAR(50),
    SubCategory VARCHAR(50),
    PaymentMode VARCHAR(20)
);

CREATE TABLE Orders (
    OrderID VARCHAR(10) PRIMARY KEY,
    OrderDate DATE,
    CustomerName VARCHAR(100),
    State VARCHAR(50),
    City VARCHAR(50)
);
